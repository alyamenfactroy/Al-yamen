document.addEventListener('DOMContentLoaded', function () {

    function adjustMegaMenuAlignment(dropdown) {
        var menu = dropdown.querySelector('.dropdown-menu');
        if (!menu) return;

        // Reset alignment so we always measure from the natural left position
        menu.classList.remove('dropdown-menu-right');

        // Use rAF to read position AFTER the browser has painted the visible menu
        requestAnimationFrame(function () {
            var rect = menu.getBoundingClientRect();
            var viewportWidth = window.innerWidth || document.documentElement.clientWidth;

            if (rect.right > viewportWidth) {
                // Overflows right edge → anchor from the right side of parent
                menu.classList.add('dropdown-menu-right');

                // Safety check: if it now overflows the left edge, revert
                var newRect = menu.getBoundingClientRect();
                if (newRect.left < 0) {
                    menu.classList.remove('dropdown-menu-right');
                }
            }
        });
    }

    document.querySelectorAll('.dropdown44.mega-menu').forEach(function (dropdown) {
        dropdown.addEventListener('mouseenter', function () {
            adjustMegaMenuAlignment(this);
        });
    });

    // Also re-check on window resize
    window.addEventListener('resize', function () {
        document.querySelectorAll('.dropdown44.mega-menu:hover').forEach(function (dropdown) {
            adjustMegaMenuAlignment(dropdown);
        });
    });

});

const DB_VERSION = 1;
const DB_NAME = site_name + "_DB";
	// Open IndexedDB
function openDatabase() {
    return new Promise((resolve, reject) => {

        let request = indexedDB.open(DB_NAME, DB_VERSION);

        request.onupgradeneeded = function (event) {
            let db = event.target.result;
            let oldVersion = event.oldVersion;

            console.log("Upgrading IndexedDB:", oldVersion, "→", DB_VERSION);

            /* ---------- PARTY STORE ---------- */
            if (!db.objectStoreNames.contains("party")) {
                let partyStore = db.createObjectStore("party", { keyPath: "id" });
                partyStore.createIndex("text_idx", "text", { unique: false });
            } else {
                let partyStore = event.target.transaction.objectStore("party");

                if (!partyStore.indexNames.contains("text_idx")) {
                    partyStore.createIndex("text_idx", "text", { unique: false });
                }
            }

            /* ---------- ITEMS STORE ---------- */
            if (!db.objectStoreNames.contains("items")) {
                let itemStore = db.createObjectStore("items", { keyPath: "id" });
                itemStore.createIndex("name_idx", "name", { unique: false });
                itemStore.createIndex("code_idx", "code", { unique: false });
            } else {
                let itemStore = event.target.transaction.objectStore("items");

                if (!itemStore.indexNames.contains("name_idx")) {
                    itemStore.createIndex("name_idx", "name", { unique: false });
                    itemStore.createIndex("code_idx", "code", { unique: false });
                }
            }

            /* ---------- META STORE (SYNC INFO) ---------- */
            if (!db.objectStoreNames.contains("meta")) {
                db.createObjectStore("meta", { keyPath: "key" });
            }
        };

        request.onsuccess = e => resolve(e.target.result);
        request.onerror = e => reject("DB error: " + e.target.error);
    });
}


async function syncPartyToIndexedDB() {
    try {
        let response = await fetch(
            "https://erp.omegasoftbd.com/erp/demo_prod/inv_suppliers/ajax_for_get_party_sync_to_IndexDB.html"
        );

        if (!response.ok) {
            throw new Error(`HTTP error! Status: ${response.status}`);
        }

        let partyList = await response.json();

        let db = await openDatabase();
        let tx = db.transaction("party", "readwrite");
        let store = tx.objectStore("party");

        /* ---------- CLEAR OLD DATA ---------- */
        await new Promise((resolve, reject) => {
            let clearReq = store.clear();
            clearReq.onsuccess = resolve;
            clearReq.onerror = reject;
        });

	async function syncItemToIndexedDB() {
		try {
			// Fetch data from server
			let response = await fetch("https://erp.omegasoftbd.com/erp/demo_prod/inv_items/ajax_for_get_item_sync_to_IndexDB.html");
			if (!response.ok) throw new Error(`HTTP error! Status: ${response.status}`);

			let item_data = await response.json();
			//console.log("Fetched item_data:", item_data);

			// Ensure we have a valid array
			if (!Array.isArray(item_data)) {
				throw new Error("Invalid response: Expected array of items from server");
			}

			// Open IndexedDB
			let db = await openDatabase();
			let transaction = db.transaction("items", "readwrite");
			let store = transaction.objectStore("items");

			// Clear old data before inserting
			let clearRequest = store.clear();

			clearRequest.onsuccess = () => {
				console.log("Existing Items data is cleared.");

				// Insert each item into IndexedDB
				item_data.forEach(item => {
					try {
						let request = store.put(item);
						request.onerror = (event) =>
							console.error("IndexedDB Insert Error:", event.target.error);
					} catch (err) {
						console.error("Error inserting item:", err, item);
					}
				});
			};

			clearRequest.onerror = (event) => console.error("Error clearing Items IndexedDB:", event.target.error);

			transaction.oncomplete = () => console.log("Items Sync complete!");
			transaction.onerror = (event) => console.error("Transaction failed:", event.target.error);

		} catch (error) {
			console.error("Item Sync failed:", error);
		}
	}


	syncItemToIndexedDB();

				}
